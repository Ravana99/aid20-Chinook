WITH
SET [~COLUMNS] AS
    Hierarchize({{[Time].[Time Hierarchy].[All Years]}, {[Time].[Time Hierarchy].[Year].Members}})
SET [~ROWS] AS
    TopCount({[Customer].[Customer Hierarchy].[Country].Members}, 10, [Measures].[Sales])
SELECT
NON EMPTY CrossJoin([~COLUMNS], {[Measures].[Sales], [Measures].[Quantity]}) ON COLUMNS,
NON EMPTY [~ROWS] ON ROWS
FROM [Sales]
